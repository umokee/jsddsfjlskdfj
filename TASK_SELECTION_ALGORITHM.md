```python
def calculate_urgency(self, task: Task, today: date) -> float:
    """Calculate task urgency for weighted random selection"""
    urgency = 0.0

    # 1. Priority component
    urgency += task.priority * 10

    # 2. Due date component
    if task.due_date:
        due_date_naive = task.due_date.replace(tzinfo=None) if task.due_date.tzinfo else task.due_date
        today_naive = datetime.combine(today, datetime.min.time())
        days_until = (due_date_naive - today_naive).days

        if days_until <= 0:
            urgency += 100  # Overdue
        elif days_until <= 2:
            urgency += 75   # Critical (‚â§2 days)
        elif days_until <= 7:
            urgency += 30   # Soon (‚â§7 days)

    # 3. Energy component
    if task.energy >= 4:
        urgency += 5
    elif task.energy <= 1:
        urgency -= 5

    return urgency
```
